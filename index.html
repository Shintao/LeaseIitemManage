async function executeSubmit(mode, data1, data2) { // 貸出モードのときだけ、両方のデータが揃っているかチェック if (mode === "生徒へ貸出") { if (!data1 || data1.trim() === "" || !data2 || data2.trim() === "") { log("🛑 [防壁作動] 貸出ですが、IDまたは端末データが空欄のためブロックしました。"); isScanning = false; if (scanOverlay) { scanOverlay.innerText = "⏸️ スキャン停止中"; scanOverlay.className = ""; } actionBtn.disabled = false; updateStatus(`⚠️ 端末QRが読み取れていません！スキャンをやり直してください。`, "#e67e22"); return; } } updateStatus("⏳ データを送信中...", "#7f8c8d"); cancelBtn.style.display = "none"; if (manualBtn) manualBtn.style.display = "none"; actionBtn.disabled = true; const formData = new FormData(); formData.append(ENTRY_MODE, mode); formData.append(ENTRY_DATA1, data1); // 🌟【ここを修正】貸出モードのときだけ data2 を送信データに含める（元々の挙動に戻す） if (mode === "生徒へ貸出" && data2 !== "") { formData.append(ENTRY_DATA2, data2); } try { await fetch(FORM_URL, { method: "POST", mode: "no-cors", body: formData }); playSound("success"); log(`✅ 送信完了しました。[モード:${mode} / D1:${data1} / D2:${data2 || "なし"}]`); if (mode === "生徒へ貸出") { updateStatus(`✅ 貸出完了！\n${savedStudentName} さん (${data1}) へ 端末 [${data2}] を登録しました。`, "#3498db"); } else if (mode === "生徒から返却") { updateStatus(`✅ 返却完了！\n端末 [${data1}] をフリーに戻しました。`, "#2ecc71"); } else if (mode.indexOf("新規") !== -1 || mode.indexOf("登録") !== -1) { updateStatus(`✅ 新規登録完了！\n端末 [${data1}] をマスターに追加。`, "#9b59b6"); } else { updateStatus(`✅ 登録解除完了！\n端末 [${data1}] を除籍しました。`, "#e67e22"); } setTimeout(() => { fetchStockInfo(); resetSystemState(); }, 3000); } catch (e) { playSound("error"); log(`❌ 送信エラー: ${e.message}`); updateStatus("❌ 送信失敗:\n" + e.message, "red"); setTimeout(() => { resetSystemState(); }, 3000); } }
